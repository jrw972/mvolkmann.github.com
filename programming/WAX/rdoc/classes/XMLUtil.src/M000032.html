  def self.escape(text)
    return text unless text.kind_of?(String)

    result = ""
    text.each_byte do |c|
      # TODO: Is using [0] the best way to do these comparisons?
      if c == '<'[0]
        result << "&lt;"
      elsif c == '>'[0]
        result << "&gt;"
      elsif c == "'"[0]
        result << "&apos;"
      elsif c == '"'[0]
        result << "&quot;"
      elsif c == '&'[0]
        result << "&amp;"
      else
        result << c
      end
    end
        
    result
  end

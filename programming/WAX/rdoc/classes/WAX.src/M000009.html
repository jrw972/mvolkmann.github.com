  def comment(text)
    # Comments can be output in any state.

    XMLUtil.verify_comment(text) if @check_me
        
    @has_content = @has_indented_content = true
    terminate_start
    write_indent if @parent_stack.size > 0

    out "<!-- #{text} -->"
    out "#{@cr}" if will_indent and @parent_stack.size == 0

    self
  end
